* new group
Criando um grupo de TI dentro da OU Matriz que já foi criada -->
#+begin_src powershell
New-ADGroup -Name "TI_Security" `
>> -GroupCategory Security `
>> -GroupScope Global `
>> -Path "OU=Matriz,DC=dc01,DC=lab" `
>> -Description "Security Team only"
#+end_src

Verificando a existência deste novo Group -->
#+begin_src powershell
 Get-ADGroup -Identity TI_Security

Criando um novo user -->
#+begin_src powershell
PS C:\Users\Administrator> $securePassword = Read-Host -AsSecureString "digite a senha"
digite a senha: *********

PS C:\Users\Administrator> New-ADUser -Name "tech user" `
>> -SamAccountName "tech.user" `
>> -UserPrincipalName "tech.user@dc01.lab" `
>> -Path "OU=Matriz,DC=dc01,DC=lab" `
>> -AccountPassword $securePassword `
>> -Enabled $true `
>> -ChangePasswordAtLogon $false
#+end_src

Adicionando usuários ao grupo security -->
#+begin_src powershell
Add-ADGroupMember -Identity "TI_Security" -Members "tech.lab"
#+end_src

Verificando todos os grupos que um usuário faz parte -->
#+begin_src powershell
Get-ADPrincipalGroupMembership -Identity "tech.lab"
#+end_src
